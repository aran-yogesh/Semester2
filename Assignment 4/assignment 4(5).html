import numpy as np
import pandas as pd 
import seaborn as sns
import matplotlib.pyplot as plt
In[2]:
tips = sns.load_dataset("tips")

plt.figure(figsize=(8, 5))

sns.violinplot(
    data=tips,  
    x="total_bill", 
    y="sex",  
    hue="sex", 
    split=True, 
    palette="pastel", 
)

plt.title("Total Bill Distribution by Gender (Violin Plot)")

plt.xlabel("Total Bill ($)")
plt.ylabel("Gender")

plt.show()

This sns violinplot visualizes the distribution and spread of total bill amounts by gender. The split view compares groups, while the inner box plot highlights quartiles and medians, making it a clear and insightful representation!
